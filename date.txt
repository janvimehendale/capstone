Date = 
ADDCOLUMNS(
    CALENDAR(DATE(2020, 1, 1), TODAY()),
    "Year", YEAR([Date]),
    "Month", MONTH([Date]),
    "MonthName", FORMAT([Date], "mmmm"),
    "Quarter", "Q" & ROUNDUP(MONTH([Date])/3, 0),
    "DayName", FORMAT([Date], "dddd"),
    "WeekNum", WEEKNUM([Date]),
    "DateKey", YEAR([Date]) * 10000 + MONTH([Date]) * 100 + DAY([Date])
)
